        function create100UniqueBooks() {
            const container = document.getElementById('books-bg');
            for (let i = 0; i < 100; i++) {
                const book = document.createElement('div');
                book.className = 'floating-book';
                book.style.backgroundImage = `url('https://covers.openlibrary.org/b/id/${uniqueBookIDs[i]}-M.jpg')`;
                book.style.left = Math.random() * 100 + 'vw';
                book.style.animationDuration = (Math.random() * 7 + 10) + 's';
                book.style.animationDelay = (Math.random() * 15) + 's';
                container.appendChild(book);
            }
        }

        
        setInterval(() => {
            const side = Math.random() > 0.5 ? 'featLeft' : 'featRight';
            const featBox = document.getElementById(side);
            const randomID = uniqueBookIDs[Math.floor(Math.random() * 100)];
            const info = bookData[Math.floor(Math.random() * bookData.length)];
            
            featBox.innerHTML = `
                <img src="https://covers.openlibrary.org/b/id/${randomID}-M.jpg">
                <h4>${info.t}</h4>
                <p>${info.d}</p>
            `;
            featBox.classList.add('active');
            setTimeout(() => featBox.classList.remove('active'), 3000);
        }, 10000);

        function showPage(id) {
            document.querySelectorAll('.page').forEach(p => p.classList.remove('active'));
            document.getElementById(id).classList.add('active');
            const books = document.getElementById('books-bg');
            const stars = document.getElementById('stars-bg');
            if (id === 'home') { books.style.display = 'block'; stars.style.display = 'none'; } 
            else { 
                books.style.display = 'none'; stars.style.display = 'block'; 
                stars.innerHTML = '';
                for(let i=0; i<80; i++) {
                    const s = document.createElement('div');
                    s.className='star'; s.style.width=s.style.height=Math.random()*4+2+'px';
                    s.style.left=Math.random()*100+'vw'; s.style.top=Math.random()*100+'vh';
                    s.style.setProperty('--d', Math.random()*3+2+'s'); stars.appendChild(s);
                }
            }
            window.scrollTo({top: 0, behavior: 'smooth'});
        }

        function toggleMusic() {
            const a = document.getElementById('bgMusic');
            const btn = document.getElementById('musicBtn');
            if (a.paused) { a.play(); btn.className = 'fas fa-pause music-icon'; document.getElementById('musicStatus').innerText = "Playing..."; }
            else { a.pause(); btn.className = 'fas fa-play music-icon'; document.getElementById('musicStatus').innerText = "Stopped"; }
        }
